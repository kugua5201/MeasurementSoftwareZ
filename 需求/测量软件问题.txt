    public event EventHandler<CacheDataEventArgs>? OnCacheDataRead;

    private CancellationTokenSource? _cacheTokenSource;

    /// <summary>
    /// 启动双缓冲区交替读取缓存数据（内部线程管理，调用者无需关心）
    /// </summary>
    public virtual void StartCacheReading(string cachAddress1, string cachLengthAddress1, string isReadCachAddress1, ushort length1, string cachAddress2, string cachLengthAddress2, string isReadCachAddress2, ushort length2)
    {
        string typeName = this.GetType().Name;
        if (!typeName.Contains("SiemensS1200PLC") && !typeName.Contains("SiemensS1500PLC"))
        {
            throw new NotSupportedException($"当前协议{typeName}不支持读取缓存数据");
        }

        StopCacheReading();

        _cacheTokenSource = new CancellationTokenSource();
        CancellationToken token = _cacheTokenSource.Token;

        _cacheThread = new Thread(() =>
        {
            CacheReadLoop(cachAddress1, cachLengthAddress1, isReadCachAddress1, length1, cachAddress2, cachLengthAddress2, isReadCachAddress2, length2, token);
        })
        {
            IsBackground = true,
            Name = "CacheReadThread"
        };
        _cacheThread.Start();
    }

    /// <summary>
    /// 停止缓存读取
    /// </summary>
    public virtual void StopCacheReading()
    {
        _cacheTokenSource?.Cancel();
        _cacheThread?.Join(3000);
        _cacheTokenSource?.Dispose();
        _cacheTokenSource = null;
        _cacheThread = null;
    }


    /// <summary>
    /// 双缓冲区交替读取主循环
    /// </summary>
    private void CacheReadLoop(string cachAddress1, string cachLengthAddress1, string isReadCachAddress1, ushort length1, string cachAddress2, string cachLengthAddress2, string isReadCachAddress2, ushort length2, CancellationToken token)
    {
        while (!token.IsCancellationRequested)
        {
            try
            {
                if (Status != DeviceStatus.Connected)
                {
                    Thread.Sleep(ConnectionArgs.PollingSpeed);
                    continue;
                }
                if (!IsOpen)
                {
                    Thread.Sleep(ConnectionArgs.PollingSpeed);
                    continue;

                }
                CacheDataEventArgs eventArgs;
                // 读取缓冲区1
                if (IsReadCach(isReadCachAddress1))
                {
                    (byte[] Data, ushort Length, bool IsSuccess, string Message) readResult = ReadCacheBytes(cachAddress1, cachLengthAddress1, length1);

                    eventArgs = new CacheDataEventArgs(1, readResult.Data, readResult.IsSuccess, readResult.Message, readResult.Length);
                    OnCacheDataRead?.Invoke(this, eventArgs);

                    if (readResult.IsSuccess)
                    {
                        WriteCacheFlag(isReadCachAddress1, false);
                    }
                }

                // 读取缓冲区2
                if (IsReadCach(isReadCachAddress2))
                {
                    (byte[] Data, ushort Length, bool IsSuccess, string Message) readResult = ReadCacheBytes(cachAddress2, cachLengthAddress2, length2);

                    eventArgs = new CacheDataEventArgs(2, readResult.Data, readResult.IsSuccess, readResult.Message, readResult.Length);
                    OnCacheDataRead?.Invoke(this, eventArgs);

                    if (readResult.IsSuccess)
                    {
                        WriteCacheFlag(isReadCachAddress2, false);
                    }
                }

                Thread.Sleep(ConnectionArgs.PollingSpeed);
            }
            catch (Exception ex)
            {

                LogError($"缓存数据读取异常: {ex.Message}");
                Thread.Sleep(ConnectionArgs.PollingSpeed);
            }
        }
    }
